function removeNode(){
　　if(isIE()||isIE11()){
　　　　obj.parentNode.removeNode(true);
　　}else{
 　　　 obj.parentNode.remove();
}
function isIE(){
　　if(!!window.ActiveXObject || "ActiveXObject" in window){
　　　　return true;
　　}else{
　　　　return false;
　　}
} 
function isIE11(){
　　if((/Trident\/7\./).test(navigator.userAgent)){
　　　　return true;
　　}else{
　　　　return false;
　　}
}
